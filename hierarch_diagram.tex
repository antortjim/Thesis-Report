\newcommand\Tunderbrace[3][]{%
  \def\tmp{#2}%
  \setbox0=\hbox{\tmp}%
  \stackunder[1pt]{%
    \stackunder[0pt]{\tmp}{\rotatebox{90}{\scaleto[2ex]{#1\{}{\wd0}}}%
  }{%
    \scriptsize #3%
  }%
}
\newcommand\Toverbrace[3][]{%
  \def\tmp{#2}%
  \setbox0=\hbox{\tmp}%
  \stackon[1pt]{%
    \stackon[0pt]{\tmp}{\rotatebox{90}{\scaleto[2ex]{#1\}}{\wd0}}}%
  }{%
    \scriptsize #3%
  }%
}
